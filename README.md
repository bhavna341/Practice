    private void jbntEqualsActionPerformed(java.awt.event.ActionEvent evt) {                                           
       NumEnter2 = Double.parseDouble(jtxtResult.getText());
       if(op == "+")
       {
           Result = NumEnter1+NumEnter2;
           jtxtResult.setText(String.valueOf(Result));
       }
       else if(op == "-")
       {
           Result = NumEnter1-NumEnter2;
           jtxtResult.setText(String.valueOf(Result));
       }
       else if(op == "*")
       {
           Result = NumEnter1*NumEnter2;
           jtxtResult.setText(String.valueOf(Result));
       }
       else if(op == "/")
       {
           Result = NumEnter1/NumEnter2;
           jtxtResult.setText(String.valueOf(Result));
       }
    }
